onTagged 
function (doclet, tag) {
    setDocletKindToTitle(doclet, tag);
    setDocletNameToValue(doclet, tag);
    if (!doclet.name) {
        setDocletNameToFilename(doclet, tag);
    }
    if (tag.value && tag.value.type) {
        doclet.type = tag.value.type;
    }
}
