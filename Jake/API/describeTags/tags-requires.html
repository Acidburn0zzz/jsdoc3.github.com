title requires
mustHaveValue true
onTagged 
function (doclet, tag) {
    var modName = firstWordOf(tag.value);
    if (modName.indexOf("module:") !== 0) {
        modName = "module:" + modName;
    }
    if (!doclet.requires) {
        doclet.requires = [];
    }
    doclet.requires.push(modName);
}
