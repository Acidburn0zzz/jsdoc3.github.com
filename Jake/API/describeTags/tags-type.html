onTagText 
function (text) {
    if (!/^\{.+\}$/.test(text)) {
        text = "{ " + text + " }";
    }
    return text;
}
onTagged 
function (doclet, tag) {
    if (tag.value && tag.value.type) {
        doclet.type = tag.value.type;
        if (doclet.kind === "function") {
            doclet.addTag("returns", tag.text);
        }
    }
}
