function (doclet, tag) {
    var ownerClassName;
    if (!tag.value) {
        ownerClassName = "{@thisClass}";
    } else {
        ownerClassName = firstWordOf(tag.value);
    }
    doclet.addTag("alias", ownerClassName);
    doclet.addTag("kind", "class");
}
