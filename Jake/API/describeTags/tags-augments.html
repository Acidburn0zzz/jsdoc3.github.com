title : augments
mustHaveValue : true
onTagText : 

function (text) {
    var type = require("jsdoc/tag/type"), tagType = type.getTagInfo(text, false, true);
    return tagType.type || text;
}


onTagged : 

function (doclet, tag) {
    doclet.augment(firstWordOf(tag.value));
}
