exception


title : exception
mustHaveValue : true
canHaveType : true
onTagged : 

function (doclet, tag) {
    if (!doclet.exceptions) {
        doclet.exceptions = [];
    }
    doclet.exceptions.push(tag.value);
    if (tag.value && tag.value.type) {
        doclet.type = tag.value.type;
    }
}


synonyms : throws
